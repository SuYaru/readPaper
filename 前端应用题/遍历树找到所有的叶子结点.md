# 遍历树找到所有的叶子结点

```javascript
const tree = {
  name: 'root',
  children: [
    {
      name: '0',
      children: [
        {
          name: '0-0'
        },
      ],
    },
    {
      name: '1',
      children: [
        {
          name: '1-0',
          children: [
            {
              name: '1-0-0',
            },
          ],
        },
      ],
    },
    {
      name: '2'
    },
  ],
}


function getLeafs(tree) {
  const leafs = [];
  function searchTree(node) {
    if (!node.children) {
      leafs.push(node);
      return;
    }
    node.children.forEach((childNode) => {
      searchTree(childNode);
    })
  }
  searchTree(tree);
  return leafs;
}
```
