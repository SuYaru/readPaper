var findContinuousSequence = function(target) {
    let sum = 1; // 总和与target相等时返回
    const moveSpace = [1]; // 滑动窗口的初始值
    const result = []; // 存储结果
    // Math.ceil 取目标值中最大的可取值
    for(let i = 2; i <= Math.ceil(target/2); i++) {
        sum += i; // 先存储最新值
        moveSpace.push(i);
        // sum 减去最小值，同时窗口移除最小值; 若一直大于，则减少到小于为止
        while(sum > target) sum -= moveSpace.shift(); 
        if (sum === target) result.push(moveSpace.slice(0))
    }
    return result
}

var findContinuousSequence = function(target) {
    const arr=[]
    for(let i=1;i<target;i++){
        arr.push(i);
    }
    let i=0;
    let win=[];
    let sum=0;
    let resArr=[];
    while(i<target-1){
        //窗口推入一个数
        win.push(arr[i]);
        //和值加上那个数
        sum+=arr[i];
        if(sum<target){
            i++;
        } else if(sum==target){
            resArr.push([...win]);
            i++;
        }else{
            //当和大于目标值
            while(sum>target){
                //缩小窗口，并减小值
                sum-=win.shift();
                if(sum==target){
                    resArr.push([...win]);
                    i++;
                }
                if(sum<target){
                    i++;
                }
            }
        }
    }
    return resArr;
};

2. 给定一个字符串S,一个字符串T，请在字符串S里面找出：包含T所有字母的最小子串[困难]

    示例：S = "ADOBECODEBANC", T = "ABC"
    输出："BANC"

```
var findMinChildStr = (s, t) => {
    let need = {}; // 需要的字符
    let moveSpace = {}; // 窗口中的字符
    for(let a of t) { need[a] = (need[a] || 0) + 1 } // 统计单个字符中出现次数

    let left = 0, right = 0, valid = 0, start = 0, len = Number.MAX_VALUE;

    while(right < s.length) {
        let c = s[right]
        right ++;
        if (need[c]) {  // 当前字符在目标字符中
            moveSpace[c] = (moveSpace[c] || 0) + 1
            if (moveSpace[c] == need[c]) {
                valid ++
            }
        }
        while(valid === Object.keys(need).length) {
            if (right - left < len) {
                start = left;
                len = right - left;
            }
            let d = s[left]
            left ++;
            if (need[d]) {
                if (moveSpace[d] === need[d]) {
                    valid --;
                }
                moveSpace[d]--;
            }
        }
    }
    return len === Number.MAX_VALUE ? "" : s.substr(start, len);
}
```

var findWordIndex = (s, p): number[] => {
    const result = [];
    const sLen = s.length;
    const pLen = p.length;
    const left = 0;
    const right = pLen;
    const newP = p.split('').sort().join(''); // a ~ z 排序
    while(right < sLen) {
        const str = s.slice(left, right).split('').sort().join('') 
        // 超出时间限制了
        if (str === newP) result.push(left)
        right ++
        left++
    }
    return result;
}
